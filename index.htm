                    hintUpdate = '精神+8，学到刺绣';﻿

                }﻿

                else {﻿

                    storyAddition = `你按照自己的心意：${actionDescription}。苗寨的山神似乎听见了你的低语。`;﻿

                    promptLine = '你的举动影响了寨子里的氛围。';﻿

                    newChoices = ['🌾 继续观察', '🗣️ 找人攀谈', '🏡 回家休息'];﻿

                    currentState.cha += 1;﻿

                    currentState.stam -= 3;﻿

                    hintUpdate = '自由行动，魅力+1';﻿

                }﻿

                // 边界限制﻿

                for (let key of ['cha','wis','phys','art','hp','stam','health','mind','money']) {﻿

                    if (currentState[key] > 100) currentState[key] = 100;﻿

                    if (currentState[key] < 0) currentState[key] = 0;﻿

                }﻿

                let newStoryHtml = `﻿

                    <p>${storyAddition}</p>﻿

                    <div class="inner-thought">✨ ${promptLine}</div>﻿

                `;﻿

                storyDisplay.innerHTML = newStoryHtml;﻿

                // 刷新选项按钮﻿

                renderChoiceButtons(newChoices);﻿

                refreshStatsUI();﻿

                gameHint.innerText = hintUpdate || '山神回应';﻿

                isApiCalling = false;﻿

                apiMessageSpan.innerText = '✅ 山神API已回应';﻿

                apiSpinner.style.display = 'none';﻿

                currentState.week += 1;﻿

            }, 1000);﻿

        }﻿
